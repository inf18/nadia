    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        nama=String.valueOf(TxtNama.getText());
        nis=String.valueOf(TxtNis.getText());
        kls=String.valueOf(TxtKelas.getText());
        smt=Integer.parseInt(TxtSemester.getText());
        
        try{
            sql="INSERT INTO siswa (nama, nis, kelas, semester) VALUE "+"('"+nama +"','"+ nis +"','"+ kls +"','"+ smt +"')";
            st=con.createStatement();
            st.execute(sql);
            Clear();
            ShowData(sql);
            JOptionPane.showMessageDialog(null,"Data Berhasil Disimpan");
                                
        }catch (Exception e){
            JOptionPane.showMessageDialog(null, "ERROR \n Data Gagal Disimpan"+e.getMessage());
        }
    }                                        
 private void Clear(){
        TxtNis.setText("");
        TxtNama.setText("");
        TxtKelas.setText("");
        TxtSemester.setText("");
    }
  private void ShowData(String sql){
        DefaultTableModel data = new DefaultTableModel();
        data.addColumn ("No");
        data.addColumn ("Nama");
        data.addColumn ("Nim");
        data.addColumn ("Kelas");
        data.addColumn ("Semester");
        try{    
            int i = 1;
            st=con.createStatement();
            siswa=st.executeQuery("select * from siswa");
            while (siswa.next())
                data.addRow(new Object[]{
                (""+i++),
                    siswa.getString(1), siswa.getString(2),
                    siswa.getString(3), siswa.getString(4),
                    siswa.getString(5)
                });
            jTable1.setModel(data);
        } catch (Exception e){
            JOptionPane.showMessageDialog(null, "ERROR \n Gagal Memuat KeDatabase \n Aktifkan Database Sebelum Memulai");
        }
    }
